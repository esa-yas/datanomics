### Design Principles
- **Dark-primary UI** — deep navy background, card surfaces, subtle borders

- **Blue = information / links** (badges, secondary actions, data)
- **Geometric angular aesthetic** — mirrors the hexagonal logo shapes; use sharp card corners (border-radius: 8px), diagonal accent lines, geometric icon badges
- **Modern data dashboard feel** — dense but clean, data-forward, professional
- **Typography**: Use `Plus Jakarta Sans` for headings, `DM Sans` for body text (import from Google Fonts)
- **No generic purple gradients, no white backgrounds** — this is a dark, navy-forward enterprise tool
- **Micro-animations**: subtle fade-ins on page load, smooth sidebar transitions, hover lifts on cards (translateY -2px + shadow)
- **Logo**: Always display the Datanomics Technology logo in the sidebar header. Use the uploaded logo image asset.

## FIREBASE SECURITY RULES

```javascript
// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    function getUserRole() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role;
    }
    function isAdmin() {
      return getUserRole() == 'admin';
    }
    function isManager() {
      return getUserRole() in ['admin', 'manager'];
    }
    function isEmployee() {
      return getUserRole() in ['admin', 'manager', 'team_lead',
        'job_search_assistant', 'resume_specialist', 'email_specialist'];
    }
    function isAssignedTo(candidateId) {
      return request.auth.uid in
        get(/databases/$(database)/documents/candidates/$(candidateId)).data.assignedTeam.values();
    }

    // Users collection
    match /users/{userId} {
      allow read: if isAuthenticated() && (request.auth.uid == userId || isManager());
      allow create: if isAdmin();
      allow update: if isAdmin() || request.auth.uid == userId;
      allow delete: if isAdmin();

      match /notifications/{notifId} {
        allow read, write: if request.auth.uid == userId;
      }
    }

    // Candidates
    match /candidates/{candidateId} {
      allow read: if isEmployee() || (getUserRole() == 'client' &&
        request.auth.uid == resource.data.uid);
      allow create: if isManager();
      allow update: if isEmployee();
      allow delete: if isAdmin();

      match /notes/{noteId} {
        allow read, write: if isEmployee();
      }
      match /followUps/{followUpId} {
        allow read, write: if isEmployee();
      }
    }

    // Applications
    match /applications/{appId} {
      allow read: if isEmployee() || (getUserRole() == 'client' &&
        request.auth.uid in get(/databases/$(database)/documents/candidates/
          $(resource.data.candidateId)).data.assignedTeam.values());
      allow create, update: if isEmployee();
      allow delete: if isManager();
    }

    // Resumes
    match /resumes/{resumeId} {
      allow read: if isEmployee();
      allow write: if isEmployee();
    }

    // Weekly Reports
    match /weeklyReports/{reportId} {
      allow read: if isEmployee() || getUserRole() == 'client';
      allow write: if isEmployee();
    }

    // Templates
    match /templates/{templateId} {
      allow read: if isEmployee();
      allow write: if isManager();
    }

    // AI Sessions - admin/manager only
    match /aiSessions/{sessionId} {
      allow read: if isManager();
      allow create: if isEmployee();
      allow update, delete: if isAdmin();
    }

    // System Settings - admin only
    match /systemSettings/{docId} {
      allow read: if isEmployee();
      allow write: if isAdmin();
    }
  }
}
```

```typescript
// src/lib/auth.ts
import { auth, db } from './firebase';
import {
  createUserWithEmailAndPassword,
  signInWithEmailAndPassword,
  signOut,
  sendPasswordResetEmail,
  onAuthStateChanged,
  User
} from 'firebase/auth';
import { doc, setDoc, getDoc, serverTimestamp } from 'firebase/firestore';

// Create new user (Admin only)
export async function createUser(
  email: string,
  password: string,
  role: UserRole,
  displayName: string,
  createdByUid: string
) {
  const credential = await createUserWithEmailAndPassword(auth, email, password);
  const uid = credential.user.uid;
  await setDoc(doc(db, 'users', uid), {
    uid,
    email,
    displayName,
    role,
    status: 'active',
    assignedCandidates: [],
    timezone: 'America/New_York',
    weeklyTargetApplications: 100,
    weeklyTargetReplies: 1,
    createdAt: serverTimestamp(),
    updatedAt: serverTimestamp(),
    createdBy: createdByUid,
  });
  return uid;
}

// Auth state observer
export function onAuthChange(callback: (user: User | null) => void) {
  return onAuthStateChanged(auth, callback);
}

// Get current user profile from Firestore
export async function getUserProfile(uid: string) {
  const snap = await getDoc(doc(db, 'users', uid));
  return snap.exists() ? snap.data() : null;
}
```

```typescript
// src/lib/ai/gemini.ts
const GEMINI_API_KEY = import.meta.env.VITE_GEMINI_API_KEY;
const GEMINI_MODEL = 'gemini-1.5-pro';
const GEMINI_BASE_URL = `https://generativelanguage.googleapis.com/v1beta/models/${GEMINI_MODEL}:generateContent?key=${GEMINI_API_KEY}`;

export async function callGemini(prompt: string, systemInstruction?: string): Promise<string> {
  const body = {
    system_instruction: systemInstruction
      ? { parts: [{ text: systemInstruction }] }
      : undefined,
    contents: [{ parts: [{ text: prompt }] }],
    generationConfig: {
      temperature: 0.7,
      maxOutputTokens: 4096,
    },
  };
  const res = await fetch(GEMINI_BASE_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(`Gemini API error: ${res.status}`);
  const data = await res.json();
  return data.candidates?.[0]?.content?.parts?.[0]?.text ?? '';
}
```

```typescript
// src/lib/ai/index.ts — Unified AI Router
import { callGemini } from './gemini';
import { callOpenAI } from './openai';
import { callClaude } from './claude';

export async function callAI(prompt: string, systemInstruction?: string): Promise<string> {
  const provider = import.meta.env.VITE_AI_PROVIDER ?? 'gemini';
  switch (provider) {
    case 'openai':    return callOpenAI(prompt);
    case 'anthropic': return callClaude(prompt);
    default:          return callGemini(prompt, systemInstruction);
  }
}
```

## BUILD INSTRUCTIONS FOR THE AI ASSISTANT

1. **Start with Firebase config and auth** — get login working first before building any UI
2. **Build the Firestore service layer** — create typed service files for each collection (candidateService.ts, applicationService.ts, etc.) before building any UI that consumes data
3. **Build layout components first** — Sidebar, TopBar, PageLayout — all pages use these
4. **Build the Candidate module** — this is the core entity everything else links to
5. **Build Application Tracker** — most important data view
6. **Build AI Resume Tailor** — highest value feature
7. **Build Manager Dashboard** — KPI aggregations from Firestore
8. **Build Client Portal** — final phase, separate route group
9. **Throughout**: use Datanomics brand colors from the CSS variables above. No white backgrounds. No purple. Dark navy everywhere. Teal for actions.
10. **Every form must have Zod validation**. Every Firestore write must have error handling. Every AI call must have a loading state and error fallback.
