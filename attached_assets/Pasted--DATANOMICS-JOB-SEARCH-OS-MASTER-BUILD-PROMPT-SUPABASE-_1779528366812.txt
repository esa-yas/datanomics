## PROJECT IDENTITY

You are building **Datanomics Job Search OS** — an internal operations platform for a professional data career placement company called **Datanomics Technology**. This is a full-stack web application with role-based access, AI-powered resume tailoring, applicant tracking, employee performance dashboards, and a client-facing portal.

This is a **production-grade** system. Build every feature as if real clients, real employees, and a technical CEO (ex-Google/Amazon/Meta Data Engineer) will review it daily.

This is a **job placement operations tool only**. There are NO students, NO courses, NO batches, NO enrollment. The candidates are job seekers paying for placement services. Keep these concepts completely separate.

### TRIGGER: Auto-increment candidate pipeline counters

```sql
CREATE OR REPLACE FUNCTION sync_candidate_pipeline()
RETURNS TRIGGER AS $$
BEGIN
  -- On new application
  IF TG_OP = 'INSERT' THEN
    UPDATE candidates SET
      total_applications = total_applications + 1,
      last_application_date = NOW(),
      updated_at = NOW()
    WHERE id = NEW.candidate_id;
  END IF;

  -- On status change to recruiter_replied
  IF TG_OP = 'UPDATE' AND NEW.status = 'recruiter_replied'
      AND OLD.status != 'recruiter_replied' THEN
    UPDATE candidates SET
      total_replies = total_replies + 1,
      last_reply_date = NOW(),
      updated_at = NOW()
    WHERE id = NEW.candidate_id;
  END IF;

  -- On status change to interview_scheduled
  IF TG_OP = 'UPDATE' AND NEW.status = 'interview_scheduled'
      AND OLD.status != 'interview_scheduled' THEN
    UPDATE candidates SET
      total_interviews = total_interviews + 1,
      last_interview_date = NOW(),
      updated_at = NOW()
    WHERE id = NEW.candidate_id;
  END IF;

  -- On status change to offer
  IF TG_OP = 'UPDATE' AND NEW.status = 'offer'
      AND OLD.status != 'offer' THEN
    UPDATE candidates SET
      total_offers = total_offers + 1,
      updated_at = NOW()
    WHERE id = NEW.candidate_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sync_pipeline
  AFTER INSERT OR UPDATE ON applications
  FOR EACH ROW EXECUTE FUNCTION sync_candidate_pipeline();
```

### TRIGGER: Log status history automatically

```sql
CREATE OR REPLACE FUNCTION log_application_status_change()
RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'INSERT' OR (TG_OP = 'UPDATE' AND NEW.status != OLD.status) THEN
    INSERT INTO application_status_history
      (application_id, status, changed_by, changed_by_name)
    VALUES
      (NEW.id, NEW.status, NEW.applied_by, NEW.applied_by_name);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_log_status
  AFTER INSERT OR UPDATE ON applications
  FOR EACH ROW EXECUTE FUNCTION log_application_status_change();
```

## ROW LEVEL SECURITY (RLS)


```typescript
import { supabase } from './supabase';
import type { AppUser } from '../types';

// Sign in
export async function signIn(email: string, password: string) {
  const { data, error } = await supabase.auth.signInWithPassword({ email, password });
  if (error) throw error;
  // Update last login
  await supabase.from('profiles')
    .update({ last_login_at: new Date().toISOString() })
    .eq('id', data.user.id);
  return data;
}

// Sign out
export async function signOut() {
  const { error } = await supabase.auth.signOut();
  if (error) throw error;
}

// Password reset
export async function resetPassword(email: string) {
  const { error } = await supabase.auth.resetPasswordForEmail(email, {
    redirectTo: `${window.location.origin}/reset-password`,
  });
  if (error) throw error;
}

// Get current user profile
export async function getMyProfile(): Promise<AppUser | null> {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return null;
  const { data } = await supabase.from('profiles').select('*').eq('id', user.id).single();
  return data;
}

// Admin creates new user (use Supabase Admin SDK in edge function for production)
export async function createUser(params: {
  email: string;
  password: string;
  displayName: string;
  role: string;
  phoneNumber?: string;
}) {
  // Step 1: Create auth user
  const { data, error } = await supabase.auth.signUp({
    email: params.email,
    password: params.password,
  });
  if (error) throw error;
  const uid = data.user!.id;
  // Step 2: Insert profile
  const { error: profileError } = await supabase.from('profiles').insert({
    id: uid,
    email: params.email,
    display_name: params.displayName,
    role: params.role,
    phone_number: params.phoneNumber ?? null,
  });
  if (profileError) throw profileError;
  return uid;
}

// Auth state observer
export function onAuthChange(callback: Parameters<typeof supabase.auth.onAuthStateChange>[0]) {
  return supabase.auth.onAuthStateChange(callback);
}
```

```typescript
      .select('id').eq('candidate_id', candidateId)
      .eq('company', company).eq('job_title', jobTitle).limit(1);

    const { data, error } = await q;

    return data;
```

```typescript
    const weekStart = new Date(now);

      candidate_id: candidateId,
      candidate_name: candidateName,
      week_start_date: weekStart.toISOString().split('T')[0],
      week_end_date: now.toISOString().split('T')[0],
      week_number: Math.ceil((now.getTime() - new Date(now.getFullYear(),0,1).getTime()) / 604800000),
      year: now.getFullYear(),
      applications_submitted: apps.length,
      recruiter_replies: replied.length,

      response_rate: apps.length > 0 ? Math.round(replied.length / apps.length * 100) : 0,
```

```typescript
async function gemini(prompt: string, system?: string): Promise<string> {
  const body: any = {
    contents: [{ parts: [{ text: prompt }] }],
    generationConfig: { temperature: 0.7, maxOutputTokens: 4096 },
  };
  if (system) body.system_instruction = { parts: [{ text: system }] };
  const res = await fetch(GEMINI_URL, {

  });
  if (!res.ok) throw new Error(`Gemini ${res.status}: ${await res.text()}`);
  const d = await res.json();
  return d.candidates?.[0]?.content?.parts?.[0]?.text ?? '';
}
```

## QUALITY SCORE UTILITY (`src/lib/utils/qualityScore.ts`)

```typescript
export function computeQualityScore(checks: {
  quality_resume_tailored: boolean;
  quality_location_verified: boolean;
  quality_salary_verified: boolean;
  quality_auth_verified: boolean;
  quality_duplicate_checked: boolean;
  quality_notes_added: boolean;
}): number {
  const weights = {
    quality_resume_tailored:  30,
    quality_location_verified: 20,
    quality_salary_verified:   15,
    quality_auth_verified:     15,
    quality_duplicate_checked: 10,
    quality_notes_added:       10,
  };
  return Object.entries(weights).reduce((score, [key, weight]) =>
    score + ((checks as any)[key] ? weight : 0), 0);
}
```
